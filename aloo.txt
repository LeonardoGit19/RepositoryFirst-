produto = float(input('Digite o valor do produto: '))
pagamento = float(input('Digite o tipo de pagamento: 1-Dinheiro 2-Cartao: '))
if pagamento == 1:
    desconto = produto - (produto * 10/100)
    print(f'Seu produto com desconto fica com total de {desconto} reais.')
elif pagamento == 2:
    desconto = float(input('Digite 1= A vista 2 = 2x 3= 3x'))
    if desconto == 1
        desconto = produto - (produto * 5/100)
        print (f'Seu produto custa o total de {desconto} reais.')
